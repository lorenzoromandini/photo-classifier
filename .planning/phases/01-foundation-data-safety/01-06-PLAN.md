       ```dart
       class TrashItem {
         final String id; // UUID
         final String originalUri;
         final String trashUri;
         final String fileName;
         final int fileSize;
         final DateTime movedAt;
         final DateTime expiresAt;
         final bool restored;
         
         const TrashItem({
           required this.id,
           required this.originalUri,
           required this.trashUri,
           required this.fileName,
           required this.fileSize,
           required this.movedAt,
           required this.expiresAt,
           this.restored = false,
         });
         
         // 7 days from movedAt
         factory TrashItem.create({
           required String originalUri,
           required String trashUri,
           required String fileName,
           required int fileSize,
         }) {
           final now = DateTime.now();
           return TrashItem(
             id: const Uuid().v4(),
             originalUri: originalUri,
             trashUri: trashUri,
             fileName: fileName,
             fileSize: fileSize,
             movedAt: now,
             expiresAt: now.add(const Duration(days: 7)),
           );
         }
       }
       ```

       - TrashCleanupWorker class with WorkManager callback
       - static const String taskName = 'trashCleanup'
       - static Future<void> callbackDispatcher():
         ```dart
         @pragma('vm:entry-point')
         static void callbackDispatcher() {
           Workmanager().executeTask((task, inputData) async {
             if (task == taskName) {
               final db = DatabaseService();
               final fileService = FileOperationService();
               final trashRepo = TrashRepository(db);
               final trashService = TrashService(fileService, trashRepo);
               
               final expiredItems = await trashRepo.getExpiredItems();
               for (final item in expiredItems) {
                 await trashService.permanentlyDelete(item);
               }
               print('Trash cleanup: deleted ${expiredItems.length} items');
             }
             return Future.value(true);
           });
         }
         ```
       
       - static Future<void> scheduleDailyCleanup():
         ```dart
         await Workmanager().registerPeriodicTask(
           taskName,
           taskName,
           frequency: const Duration(days: 1),
           constraints: Constraints(
             requiresBatteryNotLow: true,
             requiresStorageNotLow: false, // Can run even if storage low
           ),
           existingWorkPolicy: ExistingWorkPolicy.keep,
         );
         ```
