       Future<Map<String, double>> _aggregateLabels(
         List<PhotoAnalysisResult> results,
         String folderName,
       ) {
         final Map<String, _LabelStats> stats = {};
         
         // Count frequencies and sum confidences
         for (final result in results) {
           for (final label in result.labels) {
             stats.putIfAbsent(label.text, () => _LabelStats(0, 0.0));
             stats[label.text]!.count++;
             stats[label.text]!.confidenceSum += label.confidence;
           }
         }
         
         // Calculate averages
         final Map<String, double> aggregated = {};
         for (final entry in stats.entries) {
           final avgConfidence = entry.value.confidenceSum / entry.value.count;
           double score = avgConfidence;
           
           // Boost if folder name contains label
           if (folderName.toLowerCase().contains(entry.key.toLowerCase())) {
             score += 0.15; // Exact match boost
           } else if (_isPartialMatch(folderName, entry.key)) {
             score += 0.08; // Partial match boost
           }
           
           aggregated[entry.key] = score.clamp(0.0, 1.0);
         }
         
         return aggregated;
       }

       - static Future<void> callbackDispatcher() - entry point
       - static Future<void> scheduleLearning() - schedules one-time work
       
       Implementation:
       ```dart
       @pragma('vm:entry-point')
       static void callbackDispatcher() {
         Workmanager().executeTask((task, inputData) async {
           if (task == taskName) {
             final db = DatabaseService();
             final saf = SafService();
             final ml = MlLabelingService();
             final repo = LearningRepository(db, saf, ml);
             
             final pending = await repo.getPendingFolders();
             for (final folder in pending) {
               await repo.learnFolder(folder.uri);
             }
           }
           return Future.value(true);
         });
       }
       ```

    2. Update `main.dart`:
       - Initialize WorkManager in main():
         ```dart
         await Workmanager().initialize(
           callbackDispatcher,
           isInDebugMode: kDebugMode,
         );
         ```
       - Schedule learning after onboarding complete:
         ```dart
         if (isOnboardingComplete) {
           LearningWorker.scheduleLearning();
         }
         ```

    3. Schedule with constraints:
       ```dart
       static Future<void> scheduleLearning() async {
         await Workmanager().registerOneTimeTask(
           taskName,
           taskName,
           constraints: Constraints(
             networkType: NetworkType.not_required,
             requiresBatteryNotLow: true,
             requiresStorageNotLow: true,
           ),
           existingWorkPolicy: ExistingWorkPolicy.keep,
         );
       }
       ```
